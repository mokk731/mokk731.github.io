version: "2.1"
services:
  v2raya:
    image: mzz2017/v2raya:1.5.2
    container_name: v2raya
    environment:
      - PUID=1000
      - PGID=1000
      - TZ=Asia/Shanghai
      - WEBUI_PORT=2017
    volumes:
      - /etc/v2raya:/etc/v2raya
    ports:
      - 2017:2017
      - 8500-8502:8500-8502
    restart: unless-stopped
